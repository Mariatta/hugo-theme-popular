{{- /* schema.org/Event JSON-LD for an event page. Emits only resolvable keys, never empty strings/arrays/null. Time parsing follows the shared contract in PARITY.md (the same vectors drive the Astro side and the test suite). */ -}} {{- $date := .Date.Format "2006-01-02" -}} {{- $startDate := $date -}} {{- /* Leading HH:MM[am|pm] token → event-local HH:MM; else date-only. */ -}} {{- $raw := trim (.Params.time | default "") " " -}} {{- $tok := index (findRE `(?i)^[0-9]{1,2}:[0-9]{2} ?(am|pm)?` $raw 1) 0 -}} {{- with $tok -}} {{- $t := . | lower | replaceRE " " "" -}} {{- $pm := hasSuffix $t "pm" -}}{{- $am := hasSuffix $t "am" -}} {{- $hm := $t | replaceRE "(am|pm)$" "" -}} {{- $parts := split $hm ":" -}} {{- $h := int (index $parts 0) -}}{{- $min := int (index $parts 1) -}} {{- $ok := true -}} {{- if or $am $pm -}} {{- if or (lt $h 1) (gt $h 12) -}}{{- $ok = false -}}{{- end -}} {{- if and $pm (lt $h 12) -}}{{- $h = add $h 12 -}}{{- end -}} {{- if and $am (eq $h 12) -}}{{- $h = 0 -}}{{- end -}} {{- else -}} {{- if gt $h 23 -}}{{- $ok = false -}}{{- end -}} {{- end -}} {{- if gt $min 59 -}}{{- $ok = false -}}{{- end -}} {{- if $ok -}}{{- $startDate = printf "%sT%02d:%02d" $date $h $min -}}{{- end -}} {{- end -}} {{- /* Venue → location. */ -}} {{- $venuePage := "" -}} {{- with .Params.venueRef }}{{ $venuePage = site.GetPage (print "/venues/" .) }}{{ end -}} {{- /* Performers from resolvable speakers. */ -}} {{- $perf := slice -}} {{- range .Params.speakers }}{{ with site.GetPage (print "/speakers/" .) }}{{ $perf = $perf | append (dict "@type" "Person" "name" .Title) }}{{ end }}{{ end -}} {{- $currency := "USD" -}}{{- with site.Params.seo }}{{ with .currency }}{{ $currency = . }}{{ end }}{{ end -}} {{- $ld := dict "@context" "https://schema.org" "@type" "Event" "name" .Title "startDate" $startDate "eventStatus" (cond (eq .Params.cancelled true) "https://schema.org/EventCancelled" "https://schema.org/EventScheduled") "eventAttendanceMode" (cond (eq .Params.online true) "https://schema.org/OnlineEventAttendanceMode" "https://schema.org/OfflineEventAttendanceMode") "organizer" (dict "@type" "Organization" "name" site.Title "url" (absURL "/")) -}} {{- with .Description | default .Summary }}{{ $ld = merge $ld (dict "description" (. | plainify | truncate 300)) }}{{ end -}} {{- if eq .Params.online true -}} {{- $ld = merge $ld (dict "location" (dict "@type" "VirtualLocation" "url" (.Params.rsvp | default .Permalink))) -}} {{- else if $venuePage -}} {{- $loc := dict "@type" "Place" "name" $venuePage.Title -}} {{- with $venuePage.Params.address }}{{ $loc = merge $loc (dict "address" .) }}{{ end -}} {{- $ld = merge $ld (dict "location" $loc) -}} {{- end -}} {{- $img := .Params.image | default site.Params.ogImage -}} {{- with $img }}{{ $ld = merge $ld (dict "image" (. | absURL)) }}{{ end -}} {{- with $perf }}{{ $ld = merge $ld (dict "performer" .) }}{{ end -}} {{- /* Offers / pricing. */ -}} {{- if .Params.price -}} {{- $ld = merge $ld (dict "isAccessibleForFree" false "offers" (dict "@type" "Offer" "price" (printf "%v" .Params.price) "priceCurrency" (.Params.currency | default $currency) "availability" "https://schema.org/InStock" "url" .Permalink)) -}} {{- else if not .Params.cost -}} {{- $ld = merge $ld (dict "isAccessibleForFree" true "offers" (dict "@type" "Offer" "price" "0" "priceCurrency" $currency "availability" "https://schema.org/InStock" "url" .Permalink)) -}} {{- end -}}
